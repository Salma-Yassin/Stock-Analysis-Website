nameCell.appendChild(nameImage);
nameCell.appendChild(document.createTextNode(product.name));
row.appendChild(nameCell);

// Create the price cell
const priceCell = document.createElement("td");
priceCell.appendChild(document.createTextNode(product.price));
row.appendChild(priceCell);

// Create the sales cell
const salesCell = document.createElement("td");
const salesPercentage = document.createElement("small");
salesPercentage.className = `text-${product.sales.color} mr-1`;
const salesArrow = document.createElement("i");
salesArrow.className = `fas fa-arrow-${product.sales.direction}`;
const salesPercentageText = document.createTextNode(
  `${product.sales.percentage}%`
);
salesPercentage.appendChild(salesArrow);
salesPercentage.appendChild(salesPercentageText);
salesCell.appendChild(salesPercentage);
salesCell.appendChild(
  document.createTextNode(`${product.sales.total} sold`)
);
row.appendChild(salesCell);

// Create the more cell
const moreCell = document.createElement("td");
const moreLink = document.createElement("a");
moreLink.href = "#";
moreLink.className = "text-muted";
const moreIcon = document.createElement("i");
moreIcon.className = "fas fa-search";
moreLink.appendChild(moreIcon);
moreCell.appendChild(moreLink);
row.appendChild(moreCell);
